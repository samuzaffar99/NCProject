Input eq2

Output "Enter the start of the range"
Input s

Input e

Output "Enter the step"
Input h
Output "Enter the initial value"
Input i

yprev=i

Output "Modified Euler Error"

For t=s:h:e-h Do
    y0 = yprev
    k1 = h*eq(t,y0)
    k2 = h*eq(t+h,y0+k1)
    y = y0 + 0.5*(k1+k2)
    yprev = y
    y2 = eq2(t+h,y0)
    error = abs(y2-y)
    Output t+h,y,y2,error
End For
