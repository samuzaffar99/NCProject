Input eq2

Input h

Input i

yprev=i

Output "Heun Error"

For t=s:h:e-h Do
    y0 = yprev
    y = y0 + (h/4)*(eq(t,y0)+3*eq(t+(2*h/3),y0+((2*h/3)*eq(t+(h/3),y0+(h/3)*eq(t,y0)))))
    yprev = y
    y2 = eq2(t+h,y0)
    error = abs(y2-y)
    Output t+h,y,y2,error
End For
